function countWaysToReach(n, k) {
    if (n === 0) {
        return 1;
    }

    // Создаем массив для хранения количества способов для каждой клетки
    const ways = Array(n + 1).fill(0);
    
    // Устанавливаем базовое значение для клетки 0
    ways[0] = 1;

    for (let i = 1; i <= n; i++) {
        // Для каждой клетки i считаем количество способов, используя предыдущие k клеток
        for (let j = 1; j <= k; j++) {
            if (i - j >= 0) {
                ways[i] += ways[i - j];
            }
        }
    }

    return ways[n];
}
